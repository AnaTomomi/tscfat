[docs]    def list_parameters(self):
        """
        Return a dictionaty containing the key : value pairs used as function
        parameters.

        Returns
        -------
        names : dict
            key : value pairs used as function arguments.
            
        """
        names = {}
        for key, value in self.__dict__.items():
            names[key] = value
        return names
